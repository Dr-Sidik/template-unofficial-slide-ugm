\begin{frame}[t]{Syllabus (After Midterm Exam)}

    \textbf{Singular Value Decomposition (SVD)}

    \begin{itemize}
        \item SVD motivation
        \item SVD problem
        \item Geometry of SVD
        \item SVD for symmetric matrices
    \end{itemize}

    \textbf{Linear Transformations}

    \begin{itemize}
        \item The idea of linear transformation
        \item Linear transformations in matrix language
        \item Finding good basis
    \end{itemize}

    \textbf{Complex Vectors and Matrices}

    \begin{itemize}
        \item Complex number
        \item Hermitian and unitary
    \end{itemize}

\end{frame}

\begin{frame}[t]{Learning Outcomes - After Midterm Exam}
   
        \textbf{LO2} 

        Students are able to \underline{explain} the concepts of eigenvalues, eigenvectors,
        and singular value decompositions (SVD), able to \underline{compute} them.

        \vspace{0.5cm}
        \textbf{LO3} 

        Students are able to \underline{discuss} the concept of linear transformation and
        \underline{translate} this into matrix language. 
		
		\vspace{0.5cm}
        \textbf{LO4} 
		
		Students are able to \underline{do mathematical calculations} using
        complex vectors and matrices.  

\end{frame}
